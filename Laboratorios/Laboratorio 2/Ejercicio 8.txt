int Digitos (int n);


int main (){

	int n, digitos;

    	digitos=Digitos(n);

int Digitos (int n){

    int contador=1, aux=10;

    if(aux>n){

    return contador;

    }

    if(aux<=n){

        contador++;

        aux=aux*10;

        Digitos(n);

    }


}
