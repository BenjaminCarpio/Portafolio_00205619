/*Cree una matriz cuadrada de enteros que contenga cinco filas y cinco columnas e
inicialice todas sus casillas a cero. Solicite al usuario sólo aquellos valores que sean
necesarios para crear una matriz tri-diagonal y muéstrela en consola de forma
ordenada.*/

#include <iostream>
using namespace std;
int main()
{
    cout<<"Digite los valores para la matriz: "<<endl;
    int a1,a2,b1,b2,b3,c1,c2,d1,d2,d3,e1,e2;
    cin>>a1>>a2>>b1>>b2>>b3;
    cin>>c1>>c2>>d1>>d2>>d3;
    cin>>e1>>e2;
    int mat[][5] = {{a1,a2,0,0,0},{b1,b2,b3,0,0},{0,c1,0,c2,0},{0,0,d1,d2,d3},{0,0,0,e1,e2}};
    cout<<mat;
   
    for(int i=0; i<5; i++){
        for(int j=0; j<5; i++)
        cout<<mat[i][j]<<" ";
        cout<<endl;
    }
    return 0;
}
